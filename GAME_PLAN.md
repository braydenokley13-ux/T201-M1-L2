### Technical Implementation

#### Tax Calculation Logic
```javascript
function calculateLuxuryTax(payroll, taxThreshold = 165, isRepeat = false) {
  const overAmount = payroll - taxThreshold;
  if (overAmount <= 0) return 0;

  // Tax bracket multipliers
  const brackets = [
    { limit: 5, rate: isRepeat ? 1.5 : 1.0 },
    { limit: 10, rate: isRepeat ? 1.75 : 1.25 },
    { limit: 15, rate: isRepeat ? 2.0 : 1.5 },
    { limit: Infinity, rate: isRepeat ? 2.25 : 1.75 }
  ];

  let tax = 0;
  let remaining = overAmount;
  let floor = 0;

  for (const bracket of brackets) {
    const ceiling = bracket.limit === Infinity ? remaining : bracket.limit;
    const taxableInBracket = Math.min(remaining, ceiling - floor);
    tax += taxableInBracket * bracket.rate;
    remaining -= taxableInBracket;
    floor = ceiling;
    if (remaining <= 0) break;
  }

  return tax;
}
```

#### Performance Points Calculation
```javascript
function calculatePerformancePoints(wins, playoffWins, bonuses) {
  const regWinWeight = bonuses.regWinWeight || 1;
  const playoffWinWeight = bonuses.playoffWinWeight || 3;
  const fiftyWinBonus = wins >= 50 ? (bonuses.fiftyWinBonus || 5) : 0;
  const sixtyWinBonus = wins >= 60 ? (bonuses.sixtyWinBonus || 8) : 0;

  return (wins * regWinWeight) +
         (playoffWins * playoffWinWeight) +
         fiftyWinBonus +
         sixtyWinBonus;
}
```

#### Win Projection
```javascript
function projectWins(baseSituation, selectedMoves) {
  let wins = baseSituation.baseWins;
  let playoffWins = baseSituation.basePlayoffWins;
  let payroll = baseSituation.basePayroll;
  let repeatFlag = baseSituation.repeatBase;

  selectedMoves.forEach(move => {
    wins += move.winsDelta;
    playoffWins += move.playoffWinsDelta;
    payroll += move.payrollDelta;
    if (move.repeatFlag === 1) repeatFlag = 1;
  });

  return { wins, playoffWins, payroll, repeatFlag };
}
```
